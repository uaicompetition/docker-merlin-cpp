52  subindex(const variable_set& full, const variable_set& sub) {
53  assert( full >> sub);
54  m_idx = 0;
55  m_end = 1;
56  m_nd = full.nvar();
57  m_dims = full.dims();
58  m_state = new vsize[m_nd];
59  m_add = new vsize[m_nd];
60  m_subtract = new vsize[m_nd];
61  m_skipped = new bool[m_nd];
62  // Compute reference index updates
63  vsize i, j;
64  for (i = 0, j = 0; i < m_nd; ++i) {
65  m_state[i] = 1; // start with [0] = (1,1,1,1...)
66  m_skipped[i] = (j >= sub.nvar() || sub[j] != full[i]); // are we sub-indexing this variable?
67  if (i == 0)
68  m_add[i] = 1; // how much does adding one to this var
69  else
70  m_add[i] = m_add[i - 1] * (m_skipped[i - 1] ? 1 : m_dims[i - 1]); // add to our position?
71  m_subtract[i] = m_add[i] * ((m_skipped[i] ? 1 : m_dims[i]) - 1); // how much does wrapping back to 1 remove from pos?
72  if (!m_skipped[i])
73  j++;
74  m_end *= m_dims[i];
75  }
76  //end_--;
77  }

127  subindex& operator++(void) {
128  for (size_t i = 0; i < m_nd; ++i) { // for each variable
129  if (m_state[i] == m_dims[i]) { // if we reached the maximum, wrap around to 1
130  m_state[i] = 1; // subtract wrap value from position
131  if (!m_skipped[i])
132  m_idx -= m_subtract[i]; // and continue to next variable in sequence
133  } else { // otherwise, increment variable index

135  if (!m_skipped[i])

209  superindex(const variable_set& full, const variable_set& sub, size_t offset) {
210  assert( full >> sub);
211  m_idx = offset;
212  m_end = full.num_states();
213  m_ns = sub.nvar();
214  const vsize* dimf = full.dims();
215  m_dims = sub.dims();
216  m_state = new vsize[m_ns];
217  m_add = new vsize[m_ns];
218  size_t i, j, d = 1;
219  for (i = 0, j = 0; j < m_ns; ++i) {
220  if (full[i] == sub[j]) {
221  m_state[j] = 1;
222  m_add[j] = d;
223  j++;
224  }
225  d *= dimf[i];
226  }
227  m_end = m_add[m_ns - 1] * m_dims[m_ns - 1] + offset;
228  }

309  // Construct permutation mapping from VarSet -> Order
310  // (bigEndian: is 1st variable largest stride?)
314  permute_index(const std::vector<variable>& order, bool big_endian = false) {
315  m_i = 0;
316  variable_set _vs = variable_set(order.begin(), order.end(), order.size()); // compute implicit source order (VarSet)
317  m_pi.resize(order.size());
318  m_dim.resize(order.size());
319  for (size_t j = 0; j < order.size(); ++j)
320  m_dim[j] = _vs[j].states(); // save dimensions in source order (VarSet)
321  for (size_t j = 0; j < order.size(); ++j) { // compute mapping from target order to source
322  size_t jj = big_endian ? order.size() - 1 - j : j;
323  for (size_t k = 0; k < order.size(); ++k)
324  if (_vs[k] == order[j]) {
325  m_pi[jj] = k;
326  break;
327  }
328  }
329  }

349  size_t convert(size_t i) {
350  std::vector<size_t> I(m_dim.size());
351  size_t r = 0, m = 1;
352  for (size_t v = 0; v < m_dim.size(); ++v) {
353  I[v] = i % m_dim[v];
354  i -= I[v];
355  i /= m_dim[v];
356  }
357  for (size_t j = 0; j < m_dim.size(); ++j) {
358  r += m * I[m_pi[j]];
359  m *= m_dim[m_pi[j]];
360  }
361  return r;
362  }

367  permute_index inverse() {
368  permute_index inv(*this);
369  for (size_t i = 0; i < m_pi.size(); ++i) {
370  inv.m_pi[m_pi[i]] = i;
371  inv.m_dim[i] = m_dim[m_pi[i]];
372  }
373  inv.m_i = (size_t) *this;
374  return inv;
375  }

429  // BigEndian assumes that the first variable changes the fastest
430  // UAI input is assumed to follow the LittleEndian convention, whereas
431  // the internal representation of the factors assume BigEndian.

439  convert_index(const std::vector<variable>& order, bool source_big_endian = false,
440  bool target_big_endian = false) {
441 
442  m_source_order = order; // save source variable order
443  m_source_big_endian = source_big_endian; // source representation
444  m_target_big_endian = target_big_endian; // target representation
445  variable_set _vs = variable_set(order.begin(), order.end(), order.size());
446  for (variable_set::const_iterator i = _vs.begin(); i != _vs.end(); ++i) {
447  m_target_order.push_back(*i);
448  }
449  m_dim.resize(order.size());
450  for (size_t j = 0; j < order.size(); ++j) {
451  m_dim[j] = order[j].states(); // save dimensions in source order
452  }
453  }
454 
458  size_t convert(size_t i) {
459  std::vector<size_t> I(m_dim.size()); // configuration of source variables
460  if (m_source_big_endian) { // start from the first variable in order
461  for (size_t v = 0; v < m_dim.size(); ++v) {
462  I[v] = i % m_dim[v];
463  i -= I[v];
464  i /= m_dim[v];
465  }
466  } else { // start from last variable
467  for (int v = m_dim.size()-1; v >= 0; --v) {
468  I[v] = i % m_dim[v];
469  i -= I[v];
470  i /= m_dim[v];
471  }
472  }
473 
474  std::vector<size_t> J(m_dim.size()); // get configuration of target variables
475  for (size_t j = 0; j < m_target_order.size(); ++j) {
476  for (size_t k = 0; k < m_source_order.size(); ++k) {
477  if (m_target_order[j] == m_source_order[k]) {
478  J[j] = I[k];
479  break;
480  }
481  }
482  }
483 
484  // compute the index of the configuration in the target space
485  size_t r = 0, m = 1;
486  if (m_target_big_endian) {
487  for (size_t j = 0; j < m_target_order.size(); ++j) {
488  r += m*J[j];
489  m *= m_target_order[j].states();
490  }
491  } else {
492  for (int j = m_target_order.size()-1; j >= 0; --j) {
493  r += m*J[j];
494  m *= m_target_order[j].states();
495  }
496  }
497 
498  return r;
499  }
500

502  std::vector<size_t> m_dim;
503 
504  bool m_source_big_endian;
505  bool m_target_big_endian;
506  std::vector<variable> m_source_order;
507  std::vector<variable> m_target_order;
