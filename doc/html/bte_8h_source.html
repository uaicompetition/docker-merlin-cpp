486  // First downward pass to initialize the bucket tree and backward messages

488  std::cout << " - initial number of clique factors is: " << m_factors.size() << std::endl;
489  m_clusters.resize(m_order.size());

491 
492  std::cout << " - create bucket/cluster for var " << *x

494 
495  // Get the current variable to process
496  variable VX = var(*x);
497  if (*x >= vin.size() || vin[*x].size() == 0)
498  continue; // check that we have some factors over this variable
499 
500  flist ids = vin[*x]; // list of factor IDs contained in this bucket
501 
502  if (m_debug) {
503  std::cout << " - factors in this bucket: " << ids.size() << std::endl;
504  for (flist::const_iterator i = ids.begin(); i != ids.end(); ++i) {

506  }

514  size_t ii = *i;
515  if (ii == jj) continue;

540  }
541  }
542

544  vector<findex> alphas;
545  for (flistIt i = ids.begin(); i != ids.end(); ++i) {
546  //
547  // Create new cluster alpha over this set of variables; save function parameters also
548  findex alpha = findex(-1);

719  }

734 
735  factor bel = m_factors[a]; // original factors in the cluster
736 
737  // forward messages to 'a'
738  for (flist::const_iterator ci = m_in[a].begin();
739  ci != m_in[a].end(); ++ci) {
740  findex p = (*ci);
741  size_t j = m_edge_indeces[p][a];
742  bel *= m_forward[j];
743  }
744

747  ci != m_out[a].end(); ++ci) {

749  size_t j = m_edge_indeces[a][p];

802  void forward() {
803 
804  std::cout << "Begin forward (top-down) pass ..." << std::endl;
805 
806  m_logz = 0; // reset the log parition function
807  for (variable_order_t::const_iterator x = m_order.begin(); x != m_order.end(); ++x) {
808 
809  std::cout << " - Eliminating var " << *x
810  << (m_var_types[*x] ? " (MAP)\n" : " (SUM)\n");
811 
812  // Generate forward messages
813  variable VX = var(*x);
814  findex a = m_clusters[*x][0]; // get source bucket of the variable
815  if (m_out[a].size() > 0) {
816  findex b = *(m_out[a].begin()); // destination bucket
817  size_t ei = m_edge_indeces[a][b]; // edge index (for message)
818  if (m_var_types[*x] == false) { // SUM variable
819  m_forward[ei] = incoming(a, ei).sum(VX);
820  } else { // MAP variable
821  m_forward[ei] = incoming(a, ei).max(VX);
822  }
823 
824  if (m_debug) {
825  std::cout << " forward msg (" << a << "," << b << "): elim = " << VX << " -> ";
826  std::cout << m_forward[ei] << std::endl;
827  }
828  }
829 
830  } // done
831 
832  // Compute log partition function logZ or MAP/MMAP value
833  factor F(0.0);
834  for (flist::const_iterator ci = m_roots.begin();
835  ci != m_roots.end(); ++ci) {
836 
837  factor bel = calc_belief(*ci);
838  std::map<size_t, size_t>::iterator mi = m_cluster2var.find(*ci);
839  assert(mi != m_cluster2var.end());
840  size_t v = mi->second;
841  if (m_var_types[v] == false) { // SUM variable
842  F += log( bel.sum());
843  } else { // MAP variable
844  F += log( bel.max() );
845  }
846  }
847 
848  // Partition function or MAP/MMAP value
849  m_logz += F.max();
850 
851  std::cout << "Finished forward pass with logZ: " << m_logz << std::endl;
852  }

857  void backward() {
858 
859  // Calculate 'backward' messages
860  std::cout << "Begin backward (bottom-up) pass ..." << std::endl;
861 
862  vector<std::pair<findex, findex> >::reverse_iterator ri = m_schedule.rbegin();
863  for (; ri != m_schedule.rend(); ++ri ) {
864 
865  // Select backward message m(b->a)
866  findex a = (*ri).first;
867  findex b = (*ri).second;
868  size_t i = m_edge_indeces[a][b]; // edge index
869 
870  // Select variables to eliminate
871  variable_set VX = m_scopes[b] - m_separators[a][b];
872 
873  std::cout << " - Sending backward msg from " << a << " to " << b << std::endl;
874 
875  // Compute the belief at b
876  factor bel = calc_belief(b);
877  bel /= m_forward[i]; // discard the forward message on the edge
878  if (m_task == Task::PR || m_task == Task::MAR) { // marginals
879  m_backward[i] = bel.sum(VX);
880  } else if (m_task == Task::MAP) { // MAP inference
881  m_backward[i] = bel.max(VX);
882  } else { // Marginal MAP inference
883  for (variable_set::const_iterator vi = VX.begin();
884  vi != VX.end(); ++vi) {
885  size_t v = vi->label();
886  if (m_var_types[v] == false) { // SUM variable
887  bel = bel.sum(*vi);
888  } else { // MAP variable
889  bel = bel.max(*vi);
890  }
891  }
892  m_backward[i] = bel;
893  }
894 
895  if (m_debug) {
896  std::cout << " - backward msg (" << b << "," << a << "): elim = " << VX << std::endl;
897  std::cout << " -> " << m_backward[i] << std::endl;
898  }
899  }
900 
901  std::cout << "Finished backward pass." << std::endl;
902  }

908 
909  std::cout << "Begin message passing over bucket tree ..." << std::endl;
910 
911  // Top-down 'forward' message passing
912  forward();
913 
914  if (m_task == Task::PR) return; // stop here if we're only computing

916 
917  // Bottom-up 'backward' message passing
918  backward();

927  void update() {
928 
929  // update beliefs (marginals) or compute the MAP/MMAP assignment
930  if (m_task == Task::MAR) {
931  for (vindex v = 0; v < m_gmo.nvar(); ++v) {
932  findex c = m_clusters[v][0]; // get a cluster corresp. to current variable
933  variable_set vars = m_scopes[c];
934  variable VX = m_gmo.var(v);
935  variable_set out = vars - VX;
936 
937  factor bel = calc_belief(c);
938  m_beliefs[v] = marg(bel, VX);
939  m_beliefs[v] /= std::exp(m_logz); // normalize by logZ
940  }
941  } else if (m_task == Task::MAP) {
942  for (variable_order_t::const_reverse_iterator x = m_order.rbegin();
943  x != m_order.rend(); ++x) {
944 
945  //variable VX = var(*x);
946  findex a = m_clusters[*x][0]; // get source bucket of the variable
947  factor bel = incoming(a);
948 
949  // condition on previous assignment
950  for (variable_order_t::const_reverse_iterator y = m_order.rbegin();
951  y != m_order.rend(); ++y) {
952 
953  if (*y == *x) break;
954  variable VY = var(*y);
955  if (m_scopes[a].contains(VY)) {
956  bel = bel.condition(VY, m_best_config[*y]);
957  }
958  }
959  m_best_config[*x] = bel.argmax();
960  }
961  } else if (m_task == Task::MMAP) {
962  for (variable_order_t::const_reverse_iterator x = m_order.rbegin();
963  x != m_order.rend(); ++x) {
964 
965  if (m_var_types[*x] == false) break; // stop at first SUM variable
966  //variable VX = var(*x);
967  findex a = m_clusters[*x][0]; // get source bucket of the variable
968  factor bel = incoming(a);
969 
970  // condition on previous assignment
971  for (variable_order_t::const_reverse_iterator y = m_order.rbegin();
972  y != m_order.rend(); ++y) {
973  if (*y == *x) break;
974  variable VY = var(*y);
975  if (m_scopes[a].contains(VY)) {
976  bel = bel.condition(VY, m_best_config[*y]);
977  }
978  }
979  m_best_config[*x] = bel.argmax();
980  }
981  }
