493  variable_set dest(size() + B.size());
494  size_t i, j, k;
495  for (i = 0, j = 0, k = 0; i < size() && j < B.size();) {
496  if (m_v[i] == B.m_v[j]) {
497  dest.m_dlocal[k] = (m_d[i] == 0 ? B.m_d[j] : m_d[i]);
498  dest.m_v[k++] = m_v[i++];
499  j++;
500  } else if (m_v[i] > B.m_v[j]) {
501  dest.m_dlocal[k] = B.m_d[j];
502  dest.m_v[k++] = B.m_v[j++];
503  } else {
504  dest.m_dlocal[k] = m_d[i];
