287  double score(const std::vector<factor>& fin, const variable& VX, size_t i, size_t j) {
288  double err;
289  const factor& F1 = fin[i], &F2 = fin[j]; // (useful shorthand)
290  size_t iBound = std::max(std::max(m_ibound, F1.nvar() - 1),
291  F2.nvar() - 1); // always OK to keep same size
292  variable_set both = F1.vars() + F2.vars();
293  if (both.nvar() > iBound+1)
294  err = -3; // too large => -3
295  else // greedy scope-based 2 (check if useful???)
296  err = 1.0 / (F1.nvar() + F2.nvar());
297 // else // scope-based => constant score
298 // err = 1;
299  return err;

366  void tighten(size_t nIter, double stopTime = -1, double stopObj = -1) {
367 
368  std::cout << "Begin iterative cost-shifting over join graph ..." << std::endl;
369  std::cout << "+ stopObj : " << stopObj << std::endl;
370  std::cout << "+ stopTime : " << stopTime << std::endl;
371  std::cout << "+ stopIter : " << nIter << std::endl;
372 
373  double minZ = infty();
374  const std::vector<edge_id>& elist = edges();
375  double startTime = timeSystem(), dObj = infty();
376  size_t iter;
377  for (iter = 0; iter < nIter; ++iter) {
378  if (std::abs(dObj) < stopObj) {
379  break;
380  } else {
381  dObj = 0.0;
382  }
383 
384  // iterate over the join-graph edges
385  for (size_t i = 0; i < elist.size(); ++i) {
386  if (stopTime > 0 && stopTime <= (timeSystem() - startTime)) {
387  iter = nIter;
388  break;
389  }
390  findex a, b;
391  a = elist[i].first;
392  b = elist[i].second;
393  if (a > b)
394  continue;
395 
396  variable_set both = m_factors[a].vars() & m_factors[b].vars();
397  //std::cout<<_factors[a].vars()<<"; "<<_factors[b].vars()<<"= "<<both<<"\n";
398  factor fratio = (m_factors[a].maxmarginal(both)
399  / m_factors[b].maxmarginal(both)) ^ (0.5);
400  m_factors[b] *= fratio;
401  m_factors[a] /= fratio;
402  }
403 
404  // normalize the factors for numerical stability
405  for (size_t i = 0; i < num_factors(); ++i) {
406  double maxf = m_factors[i].max();
407  m_factors[i] /= maxf;
408  double lnmaxf = std::log(maxf);
409  m_log_z += lnmaxf;
410  dObj -= lnmaxf;
411  }
412 
413  // keep track of the tightest upper bound (and configuration)
414  if (m_log_z < minZ) {
415  minZ = m_log_z;
416  m_best_config = config();
417  }
418 
419  std::cout << " JGLP: " << std::fixed << std::setw(12) << std::setprecision(6)
420  << m_log_z << " (" << std::scientific << std::setprecision(6)
421  << std::exp(m_log_z) << ") ";
422  std::cout << "\td=" << dObj << "\t time=" << std::fixed
423  << std::setprecision(6) << (timeSystem() - m_start_time)
424  << "\ti=" << iter << std::endl;
425 
426  } // done
427 
428  // final logZ is the tightest one
429  m_log_z = minZ;
430 
431  double Zdist = std::exp(m_log_z / num_factors());
432  for (size_t f = 0; f < num_factors(); ++f)
433  m_factors[f] *= Zdist; // !!!! WEIRD; FOR GLOBAL CONSTANT TRANFER
434 
435  // Output solution (UAI output format)
436  std::cout << "Converged after " << iter << " iterations in "
437  << (timeSystem() - m_start_time) << " seconds" << std::endl;
438  }

470  void run() {
471 
472  // Initialize
473  init();
474 
475  // Prepare the buckets
476  std::vector<factor> fin(m_gmo.get_factors()); // get the original input factors
477  std::vector<double> Norm(m_gmo.num_factors(), 0.0); // and normalize them
478  for (size_t i = 0; i < m_gmo.num_factors(); ++i) {
479  double mx = fin[i].max();
480  fin[i] /= mx;
481  Norm[i] = std::log(mx);
482  m_log_z += Norm[i];
483  }
484 
485  // Map each variable to the list of factors mentioning that variable
486  std::vector<flist> vin;
487  for (size_t i = 0; i < m_gmo.nvar(); ++i) {
488  vin.push_back(m_gmo.with_variable(var(i)));
489  }
490 
491  // Origination info: which original factors are included for the first
492  // time, and which newly created clusters feed into this cluster
493  std::vector<flist> Orig(m_gmo.num_factors());
494  std::vector<flist> New(m_gmo.num_factors());
495  for (size_t i = 0; i < Orig.size(); ++i) {
496  Orig[i] |= i;
497  }
498 
499  // save the mini-buckets (as lists of factor indeces)
500  m_mini_buckets.resize(m_gmo.nvar());
501 
502  // Eliminate each variable in the sequence given:
503  if (m_debug) std::cout << "Initialize join graph ..." << std::endl;
504 
505  for (variable_order_t::const_iterator x = m_order.begin(); x != m_order.end(); ++x) {
506 
507  if (m_debug) std::cout << " Eliminating variable " << *x << std::endl;
508 
509  variable VX = var(*x);
510  if (*x >= vin.size() || vin[*x].size() == 0)
511  continue; // check that we have some factors over this variable
512 
513 
514  // list of factor IDs contained in this bucket
515  flist ids = vin[*x];
516 
517  // partition into mini-buckets
518  partition(VX, fin, vin, Norm, Orig, New, ids);
519 
520  // Perform any matching?
521  // "Matching" here is: compute the largest overlap of all buckets,
522  // and ensure that the
523  // moments on that subset of variables are identical in all buckets.
524  if ( ids.size() > 1 ) {
525  std::vector<factor> ftmp(ids.size()); // compute geometric mean
526  variable_set var = fin[ids[0]].vars(); // on all mutual variables
527  for (size_t i = 1; i < ids.size(); i++)
528  var &= fin[ids[i]].vars();
529  //Factor fmatch(var,1.0);
530  factor fmatch(var, 0.0);
531  for (size_t i = 0; i < ids.size(); i++) {
532  //ftmp[i] = marg(fin[ids[i]],var);
533  //fmatch *= ftmp[i];
534  ftmp[i] = marg(fin[ids[i]], var).log();
535  fmatch += ftmp[i];
536  }
537  //fmatch ^= (1.0/ids.size()); // and match each bucket to it
538  fmatch *= (1.0 / ids.size()); // and match each bucket to it
539  //for (size_t i=0;i<ids.size();i++) fin[ids[i]] *= fmatch/ftmp[i];
540  for (size_t i = 0; i < ids.size(); i++)
541  fin[ids[i]] *= (fmatch - ftmp[i]).exp();
542  }
543 
544  // Eliminate individually within buckets
545  std::vector<findex> alphas;
546  for (flistIt i = ids.begin(); i != ids.end(); ++i) {
547 
548  // Create new cluster alpha over this set of variables;
549  // save function parameters also
550  findex alpha = findex(-1);//, alpha2 = findex(-1);
551  alpha = add_factor(fin[*i]);
552  alphas.push_back(alpha);
553  m_mini_buckets[*x] |= alpha;
554 
555  fin[*i] = elim(fin[*i], VX);
556 
557  m_factors[alpha] /= fin[*i];
558 
559  // normalize for numerical stability
560  double maxf = fin[*i].max();
561  fin[*i] /= maxf;
562  maxf = std::log(maxf);
563  m_log_z += maxf;
564  Norm[*i] += maxf; // save normalization for overall bound
565 
566  for (size_t j = 0; j < alphas.size() - 1; ++j)
567  add_edge(alpha, alphas[j]);
568  for (flistIt j = New[*i].begin(); j != New[*i].end(); ++j)
569  add_edge(*j, alpha);
570 
571  Orig[*i].clear();
572  New[*i].clear();
573  New[*i] |= alpha; // now incoming nodes to *i is just alpha
574 
575  insert(vin, *i, fin[*i].vars()); // recompute and update adjacency
576  }
577  }
579 
580  if (m_debug) {
581  std::cout << "Finished creating the join graph." << std::endl;
582  }
583 
584  // Compute the initial upper bound
585  factor F(0.0);
586  for (size_t i = 0; i < fin.size(); ++i)
587  F += log(fin[i]);
588  assert(F.nvar() == 0);
589  m_log_z += F.max();
590 
591  std::cout << "Initial Upper Bound is " << std::fixed << std::setw(12)
592  << std::setprecision(MERLIN_DOUBLE_PRECISION)
593  << m_log_z << " (" << std::scientific
594  << std::setprecision(MERLIN_DOUBLE_PRECISION)
595  << std::exp(m_log_z) << ")" << std::endl;
596 
597  // Followed by iterative cost-shifting tighteing via JG propagation
598  tighten(m_num_iter);
599 
600  // Output the MAP assignment
601  std::cout << "Final Upper Bound is " << std::fixed << std::setw(12)
602  << std::setprecision(MERLIN_DOUBLE_PRECISION)
603  << m_log_z << " (" << std::scientific
604  << std::setprecision(MERLIN_DOUBLE_PRECISION)
605  << std::exp(m_log_z) << ")" << std::endl;
606 
607  m_lb = m_gmo.logP(m_best_config);
608  std::cout << "Final Lower Bound is " << std::fixed << std::setw(12)
609  << std::setprecision(MERLIN_DOUBLE_PRECISION)
610  << m_lb << " (" << std::scientific
611  << std::setprecision(MERLIN_DOUBLE_PRECISION)
612  << std::exp(m_lb) << ")" << std::endl;
613 
614  std::cout << "MAP" << std::endl;
615  std::cout << m_best_config.size() << " ";
616  std::copy(m_best_config.begin(), m_best_config.end(),
617  std::ostream_iterator<index>(std::cout, " "));
618  std::cout << std::endl;
619  }

656  // Partition the bucket into mini-buckets; initially 'ids' is the bucket,
657  // but at the end it contains a list of mini-buckets (factors in each
658  // mini-bucket are combined into a single one, thus a mini-bucket is a factor
659  // defined over at most iBound distinct variables).
660  void partition(variable VX, std::vector<factor>& fin, std::vector<flist>& vin,
661  std::vector<double>& Norm, std::vector<flist>& Orig,
662  std::vector<flist>& New, flist& ids) {
663 
664  // Select allocation into buckets
665  typedef std::pair<double, sPair> _INS;
666  std::multimap<double, sPair> scores;
667  std::map<sPair, std::multimap<double, sPair>::iterator> reverseScore;
668 
669 // std::cout<<"Initial table sizes: ";
670 // for (flistIt i=ids.begin();i!=ids.end();++i)
671 // std::cout<<fin[*i].numel()<<" "; std::cout<<"\n";
672 // std::cout << "Initial factors: ";
673 // for (flistIt i=ids.begin();i!=ids.end();++i)
674 // std::cout<<*i << ": " <<fin[*i]<<"; "; std::cout<<"\n";
675 
676  // Populate list of pairwise scores for aggregation
677  for (flistIt i = ids.begin(); i != ids.end(); ++i) {
678  for (flistIt j = ids.begin(); j != i; ++j) {
679  double err = score(fin, VX, *i, *j);
680  sPair sp(*i, *j);
681  reverseScore[sp] = scores.insert(_INS(err, sp)); // save score
682  }
683  reverseScore[sPair(*i, *i)] = scores.insert(
684  _INS(-1, sPair(*i, *i))); // mark self index at -1
685  }
686 
687  // Run through until no more pairs can be aggregated:
688  // Find the best pair (ii,jj) according to the scoring heuristic
689  // and join them as jj; then remove ii and re-score all pairs with jj
690  for (;;) {
691  std::multimap<double, sPair>::reverse_iterator top = scores.rbegin();
692  if (top->first < 0)
693  break; // if can't do any more, quit
694  else {
695  size_t ii = top->second.first, jj = top->second.second;
696 // std::cout<<"Joining "<<ii<<","<<jj<<"; size "<<(fin[ii].vars()+fin[jj].vars()).nrStates()<<"\n";
697  fin[jj] *= fin[ii]; // combine into j
698  Norm[jj] += Norm[ii];
699  double mx = fin[jj].max();
700  fin[jj] /= mx;
701  mx = std::log(mx);
702  m_log_z += mx;
703  Norm[jj] += mx;
704  erase(vin, ii, fin[ii].vars());
705  fin[ii] = factor(); // & remove i
706 
707  Orig[jj] |= Orig[ii];
708  Orig[ii].clear(); // keep track of list of original factors in this cluster
709  New[jj] |= New[ii];
710  New[ii].clear(); // list of new "message" clusters incoming to this cluster
711 
712  for (flistIt k = ids.begin(); k != ids.end(); ++k) { // removing entry i => remove (i,k) for all k
713  scores.erase(reverseScore[sPair(ii, *k)]);
714  }
715 
716  ids /= ii;
717 
718  for (flistIt k = ids.begin(); k != ids.end(); ++k) { // updated j; rescore all pairs (j,k)
719  if (*k == jj)
720  continue;
721  double err = score(fin, VX, jj, *k);
722  sPair sp(jj, *k);
723  scores.erase(reverseScore[sp]); // change score (i,j)
724  reverseScore[sp] = scores.insert(_INS(err, sp)); //
725  }
726  }
727  }
728  }
