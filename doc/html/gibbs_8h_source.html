158  void run() {
159 
160  init();
161 
162  std::cout << "Initial score: " << m_lb << std::endl;
163  std::cout << "Initial state: ";
164  std::copy(m_state.begin(), m_state.end(),
165  std::ostream_iterator<size_t>(std::cout, " "));

168  // Generate the samples
169  double score = m_lb;
170  for (size_t j = 0, i = 0; i < m_num_samples; ++i) {// keep nSamples evenly space samples
171  size_t jNext = (size_t) ((1.0 + i) / m_num_samples * m_num_iter); // among nIter steps
172  for (; j < jNext; ++j) {
173 
174  // Each iteration, go over all the variables:
175  std::vector<index> sample(nvar(), -1); // new sample to be generated
176  for (size_t v = 0; v < nvar(); ++v) {
177  assert(var(v).states() != 0); // (make sure they're non-empty)
178 
179  const flist& factors = with_variable(var(v)); // get the factors they are involved with
180  factor F(var(v), 1.0);
181  for (flist::const_iterator f = factors.begin();
182  f != factors.end(); ++f) {
183  variable_set vs = get_factor(*f).vars();
184  vs /= var(v); // and condition on their neighbors
185  F *= get_factor(*f).slice(vs, sub2ind(vs, m_state));
186  }
187 
188  score -= std::log(F[m_state[v]]);// remove current value
189  if (m_temp != 1.0) {
190  sample[v] = (F ^ m_temp).sample(); // then draw a new value
191  } else {
192  sample[v] = F.sample(); // (annealed or not)
193  }
194 
195  score += std::log(F[sample[v]]);// update score incrementally
196  } // end iterating over each variable
197 
198  if (score > m_lb) {
199  m_lb = score;
200  m_best_config = sample;
201  } // and keep the best
202 
203  // We have a new sample
204  m_state = sample;
205 
206  // After each sweep, update our statistics
207  if (isinf(score)) {// if we're still looking for a valid config
208  score = 0.0; // we need to update the score completely
209  for (size_t f = 0; f < num_factors(); ++f)
210  score += std::log(
211  get_factor(f)[sub2ind(get_factor(f).vars(), m_state)]);
212  }
213 // for (size_t v = 0; v < nvar(); ++v) { // then run through the factors
214 // m_beliefs[v][m_state[v]] += 1.0 / m_num_samples; // and update their status
215 // }
216 
217  if (m_temp_min != m_temp_max)
218  m_temp += (m_temp_max - m_temp_min) / m_num_iter;// update temperature if annealed
219 
220  }
221 
222  m_samples.push_back(m_state);
223  }
224 
225  // check out the samples
226  if (m_debug) {
227  std::cout << "Samples generated: " << m_samples.size() << std::endl;
228  for (size_t s = 0; s < m_samples.size(); ++s) {
229  std::copy(m_samples[s].begin(), m_samples[s].end(),
230  std::ostream_iterator<index>(std::cout, " "));
231  std::cout << std::endl;
232  }
233  }
234 
235  // update the beliefs
236  for (size_t v = 0; v < nvar(); ++v) { // then run through the factors
237  for (size_t s = 0; s < m_samples.size(); ++s) {
238  std::vector<index>& state = m_samples[s];
239  m_beliefs[v][state[v]] += 1.0;
240  }
241  m_beliefs[v] /= m_num_samples;
242  }
