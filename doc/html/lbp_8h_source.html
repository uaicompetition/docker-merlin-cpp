324  // Run algorithm LBP
325  void run() {
326 
327  init();
328 
329  // it's easier to count updates than "iterations"
330  size_t stopIter = m_stop_iter * num_factors();
331 
332  double dObj = m_stop_obj + 1.0, dMsg = m_stop_msg + 1.0;// initialize termination values
333  size_t iter = 0, print = 1; // count updates and "iterations" for printing
334  size_t f, n = 0;
335 
336  std::cout << "Begin message passing over factor graph ..." << std::endl;
337  std::cout << "+ stopObj : " << m_stop_obj << std::endl;
338  std::cout << "+ stopMsg : " << m_stop_msg << std::endl;
339  std::cout << "+ stopIter : " << stopIter << std::endl;
340  for (; dMsg >= m_stop_msg && iter < stopIter && dObj >= m_stop_obj;) {
341 
342  if (m_sched == Schedule::Priority) { // priority schedule =>
343  f = edge(m_priority.top().second).second; // get next factor for update from queue
344  m_priority.pop();
345  } else { // fixed schedule =>
346  f = m_forder[n]; // get next factor from list
347  if (++n == m_forder.size()) {
348  n = 0; // loop over to the beginning of the fixed order
349  }
350  }
351 
352  if (m_sched != Schedule::Flood) {// For non-"flood" schedules,
353  factor logF = log(get_factor(f));
354  dObj = 0.0; // compute new belief and update objective:
355  dObj -= (belief(f) * logF).sum() + obj_entropy(f); // remove old contribution
356  accept_incoming(f); // accept all messages into factor f
357  m_log_z += dObj += (belief(f) * logF).sum() + obj_entropy(f); // re-add new contribution
358  }
359  update_outgoing(f); // update outgoing messages from factor f
360 
361  if (m_sched == Schedule::Priority) {
362  dMsg = m_priority.top().first; // priority schedule => easy to check msg stop
363  } else if (m_stop_msg > 0 && n == 0) { // else check once each time through all factors
364  dMsg = 0.0;
365  for (size_t e = 0; e < 2 * num_edges(); ++e) {
366  dMsg = std::max(dMsg, m_msg_new[e].distance(m_msg[e], m_dist));
367  }
368  }
369 
370  if (m_sched == Schedule::Flood && n == 0) { // for flooding schedules, recalculate all
371  dObj = m_log_z;
372  m_log_z = 0.0; // the beliefs and objective now
373  for (size_t f = 0; f < num_factors(); ++f) {
374  accept_incoming(f);
375  m_log_z += (belief(f) * log(get_factor(f))).sum() + obj_entropy(f);
376  }
377  dObj -= m_log_z;
378  }
379 
380  if (iter > print * num_factors()) {
381  print++;
382  std::cout << " LBP: " << std::fixed << std::setw(12)
383  << std::setprecision(MERLIN_DOUBLE_PRECISION)
384  << m_log_z << " (" << std::scientific
385  << std::setprecision(MERLIN_DOUBLE_PRECISION)
386  << std::exp(m_log_z) << ") ";
387  std::cout << "\td=" << std::scientific
388  << std::setprecision(MERLIN_DOUBLE_PRECISION)
389  << dObj << "\tm=" << dMsg << "\t time=" << std::fixed
390  << std::setprecision(MERLIN_DOUBLE_PRECISION)
391  << (timeSystem() - m_start_time)
392  << "\ti=" << iter << std::endl;
393  }
394 
395  iter++;
396  }
397 
398  // Output solution (UAI output format)
399  std::cout << "Converged after " << iter << " iterations in "
400  << (timeSystem() - m_start_time) << std::endl;
401  std::cout << "PR" << std::endl;
402  std::cout << std::fixed << std::setprecision(MERLIN_DOUBLE_PRECISION)
403  << m_log_z << " (" << std::scientific
404  << std::setprecision(MERLIN_DOUBLE_PRECISION)
405  << std::exp(m_log_z) << ")" << std::endl;
406  std::cout << "MAR" << std::endl;
407  std::cout << nvar();
408  for (size_t v = 0; v < m_vindex.size(); ++v) {
409  variable VX = var(v);
410  std::cout << " " << VX.states();
411  for (size_t j = 0; j < VX.states(); ++j) {
412  std::cout << " " << std::fixed
413  << std::setprecision(MERLIN_DOUBLE_PRECISION) << belief(VX)[j];
414  }
415  }
416  std::cout << std::endl;
417

419  std::cout << "Final log partition function is " << m_log_z << std::endl;
420  std::cout << "Final (normalized) beliefs\n";
421  for (size_t i = 0; i < m_beliefs.size(); ++i) {
422  std::cout << m_beliefs[i] << std::endl;
423  }
424  }

462  void calc_belief(size_t n) {
463  const set<edge_id>& nbrs = neighbors(n); // get all incoming edges
464  bel(n) = factor(n); // calculate local factor times messages
465  for (set<edge_id>::const_iterator i = nbrs.begin(); i != nbrs.end(); ++i)
466  bel(n) *= m_msg[i->ridx];
467  bel(n) /= bel(n).sum(); // and normalize
468  }
469 
476  void accept_incoming(size_t n) {
477  const set<edge_id>& nbrs = neighbors(n); // get the list of neighbors
478  bel(n) = get_factor(n); // and start with just the local factor
479  for (set<edge_id>::const_iterator i = nbrs.begin();
480  i != nbrs.end(); ++i) {
481  m_msg[i->ridx] = m_msg_new[i->ridx]; // accept each new incoming message
482  bel(n) *= m_msg[i->ridx]; // and include it in the belief
483  if (m_sched == Schedule::Priority) {
484  m_priority.erase(i->ridx); // accepted => remove from priority_ queue
485  }
486  }
487  bel(n) /= bel(n).sum(); // normalize belief
488  }
489 
496  void update_outgoing(size_t n) {
497  const set<edge_id>& nbrs = neighbors(n); // get the list of neighbors
498  for (set<edge_id>::const_iterator i = nbrs.begin();
499  i != nbrs.end(); ++i) {
500  m_msg_new[i->idx] = (belief(n) / m_msg[i->ridx]).marginal(belief(i->second).vars());
501  m_msg_new[i->idx] /= m_msg_new[i->idx].sum(); // normalize message
502  if (m_sched == Schedule::Priority) { // and update m_priority in schedule
503  m_priority.insert(m_msg_new[i->idx].distance(m_msg[i->idx], m_dist),
504  i->idx);
505  }
506  }
507  }
