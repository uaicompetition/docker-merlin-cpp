39  // Select the inference task

44  if (strcasecmp(input_task, "PR") == 0) {

47  task = MERLIN_TASK_MAR;
48  } else if (strcasecmp(input_task, "MPE") == 0) {
49  task = MERLIN_TASK_MAP;

51  task = MERLIN_TASK_MMAP;

55  exit(1);

58  // Setup Merlin engine
