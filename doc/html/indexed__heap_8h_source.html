101  void insert(double p, size_t r) {
102  if (m_rev.size() <= r)
103  m_rev.resize(r + 1, 0);
104  size_t i;
105  if (m_rev[r] != 0) {
106  i = m_rev[r];
107  m_p[i - 1] = p;
108  max_heapify(i);
109  } else {
110  m_p.push_back(p);
111  m_id.push_back(r);
112  m_rev[r] = size();
113  i = size();
114  }
115  for (;;) {
116  size_t parent = i / 2;
117  if (parent > 0 && m_p[parent - 1] < m_p[i - 1]) {
118  heap_swap(parent, i);
119  i = parent;
120  } else
121  return;
122  }
123  }

208  void max_heapify(size_t i) {
209  for (;;) {
210  size_t left = 2 * i, right = 2 * i + 1, largest = i;
211  if (left <= m_p.size() && m_p[left - 1] > m_p[largest - 1])
212  largest = left;

214  largest = right;
