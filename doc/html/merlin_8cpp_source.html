135 bool Merlin::read_evidence(const char* filename) {
136  try {
137 
138  // Open the evidence file
139  std::ifstream in(filename);
140  if (in.fail()) {
141  throw std::runtime_error("Error while opening the evidence file.");
142  }
143 
144  // Clear any previous evidence
145  m_evidence.clear();
146 
147  // Read the evidence file
148  int num_evid;
149  in >> num_evid;
150  for (int i = 0; i < num_evid; ++i) {
151  vindex var;
152  size_t val;
153  in >> var >> val;
154  m_evidence[var] = val;
155  }
156 
157  // Close the evidence file
158  in.close();
159 
160  return true;
161  } catch (const std::runtime_error& e) {
162  std::cerr << e.what() << std::endl;
163  return false;
164  }
165 }

224 void Merlin::check() {
225  if (m_task == MERLIN_TASK_PR) {
226  if (m_algorithm != MERLIN_ALGO_WMB &&
227  m_algorithm != MERLIN_ALGO_BE) {
228  throw std::runtime_error("Unsupported PR inference algorithm. Use WMB, BE.");
229  }
230  } else if (m_task == MERLIN_TASK_MAR) {
231  if (m_algorithm != MERLIN_ALGO_WMB &&
232  m_algorithm != MERLIN_ALGO_IJGP &&
233  m_algorithm != MERLIN_ALGO_LBP &&
234  m_algorithm != MERLIN_ALGO_GIBBS &&
235  m_algorithm != MERLIN_ALGO_BE) {
236  throw std::runtime_error("Unsupported MAR inference algorithm. Use WMB, IJGP, LBP, GIBBS.");
237  }
238  } else if (m_task == MERLIN_TASK_MAP) {
239  if (m_algorithm != MERLIN_ALGO_WMB &&
240  m_algorithm != MERLIN_ALGO_JGLP &&
241  m_algorithm != MERLIN_ALGO_IJGP &&
242  m_algorithm != MERLIN_ALGO_GIBBS &&
243  m_algorithm != MERLIN_ALGO_BE) {
244  throw std::runtime_error("Unsupported MAP inference algorithm. Use WMB, JGLP, IJGP, GIBBS.");
245  }
246  } else if (m_task == MERLIN_TASK_MMAP) {
247  if (m_algorithm != MERLIN_ALGO_WMB &&
248  m_algorithm != MERLIN_ALGO_BE) {
249  throw std::runtime_error("Unsupported MMAP inference algorithm. Use WMB.");
250  }
251  } else {
252  throw std::runtime_error("Unsupported inference task. Use PR, MAR, MAP, MMAP");
253  }

259 int Merlin::run() {
260 
261  try {
262 
263  // Safety checks
264  check();
265 
266  // Initialize the graphical model
267  merlin::graphical_model gm;
268  std::vector<merlin::factor> fs;
269  std::map<vindex, vindex> old2new;
270  gm = *(static_cast<merlin::graphical_model*>(m_gmo));
271 
272  // Assert the evidence
273  if ( m_evidence.empty() == false ) {
274  fs = gm.assert_evidence( m_evidence, old2new );
275  } else {
276  fs = gm.get_factors();
277  for (size_t v = 0; v < gm.nvar(); ++v) {
278  old2new[v] = v;
279  }
280  }
281 
282  // Setup the solver to run
283  if ( m_task == MERLIN_TASK_PR ) {
284  size_t found = m_filename.find_last_of("/");
285  std::string prob_name = (found != std::string::npos) ?
286  m_filename.substr(found + 1) : m_filename;
287  std::string output_file = "./" + prob_name + ".PR";
288 
289  if (m_algorithm == MERLIN_ALGO_WMB) {
290  merlin::wmb s(fs);
291  std::ostringstream oss;
292  oss << "iBound=" << m_param_ibound << ","
293  << "Order=MinFill" << ","
294  << "OrderIter=100" << ","
295  << "Iter=" << m_param_iterations << ","
296  << "Task=PR,Debug=0";
297  s.set_properties(oss.str());
298  s.run();
299  s.write_solution(output_file.c_str(), m_evidence, old2new, gm, m_first_solution);
300  } else if (m_algorithm == MERLIN_ALGO_BE) {
301  merlin::bte s(fs);
302  std::ostringstream oss;
303  oss << "Order=MinFill" << ","
304  << "Task=PR";
305  s.set_properties(oss.str());
306  s.run();
307  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
308  }
309  } else if ( m_task == MERLIN_TASK_MAR ) {
310  size_t found = m_filename.find_last_of("/");
311  std::string prob_name = (found != std::string::npos) ?
312  m_filename.substr(found + 1) : m_filename;
313  std::string output_file = "./" + prob_name + ".MAR";
314 
315  if (m_algorithm == MERLIN_ALGO_WMB) {
316  merlin::wmb s(fs);
317  std::ostringstream oss;
318  oss << "iBound=" << m_param_ibound << ","
319  << "Order=MinFill" << ","
320  << "OrderIter=100" << ","
321  << "Iter=" << m_param_iterations << ","
322  << "Task=MAR";
323  s.set_properties(oss.str());
324  s.run();
325  s.write_solution(output_file.c_str(), m_evidence, old2new, gm, m_first_solution);
326  } else if (m_algorithm == MERLIN_ALGO_IJGP) {
327  merlin::ijgp s(fs);
328  std::ostringstream oss;
329  oss << "iBound=" << m_param_ibound << ","
330  << "Order=MinFill" << ","
331  << "Iter=" << m_param_iterations << ","
332  << "Task=MAR";
333  s.set_properties(oss.str());
334  s.run();
335  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
336  } else if (m_algorithm == MERLIN_ALGO_LBP) {
337  merlin::lbp s(fs);
338  std::ostringstream oss;
339  oss << "Schedule=Fixed,Distance=HPM,StopIter="
340  << m_param_iterations << ",StopObj=-1,StopMsg=-1,Debug=0";
341  s.set_properties(oss.str());
342  s.run();
343  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
344  } else if (m_algorithm == MERLIN_ALGO_GIBBS) {
345  merlin::gibbs s(fs);
346  std::ostringstream oss;
347  oss << "Task=MAR,TempMin=1.0,TempMax=1.0,Debug=0" << ","
348  << "Iter=" << m_param_iterations << ","
349  << "Samples=" << m_param_samples;
350  s.set_properties(oss.str());
351  s.run();
352  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
353  } else if (m_algorithm == MERLIN_ALGO_BE) {
354  merlin::bte s(fs);
355  std::ostringstream oss;
356  oss << "Order=MinFill" << ","
357  << "Task=MAR,Debug=1";
358  s.set_properties(oss.str());
359  s.run();
360  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
361  }
362  } else if ( m_task == MERLIN_TASK_MAP ) {
363  size_t found = m_filename.find_last_of("/");
364  std::string prob_name = (found != std::string::npos) ?
365  m_filename.substr(found + 1) : m_filename;
366  std::string output_file = "./" + prob_name + ".MPE";
367 
368  if (m_algorithm == MERLIN_ALGO_WMB) {
369  merlin::wmb s(fs);
370  std::ostringstream oss;
371  oss << "iBound=" << m_param_ibound << ","
372  << "Order=MinFill" << ","
373  << "OrderIter=100" << ","
374  << "Iter=" << m_param_iterations << ","
375  << "Task=MAP";
376  s.set_properties(oss.str());
377  std::vector<vindex> qvars;
378  for (size_t i = 0; i < gm.nvar(); ++i) {
379  if (m_evidence.find(i) == m_evidence.end()) {
380  size_t nvar = old2new.at(i);
381  qvars.push_back(nvar); // use the new index of the MAP vars
382  }
383  }
384  s.set_query(qvars);
385  s.run();
386  s.write_solution(output_file.c_str(), m_evidence, old2new, gm, m_first_solution);
387  } else if (m_algorithm == MERLIN_ALGO_JGLP) {
388  merlin::jglp s(fs);
389  std::ostringstream oss;
390  oss << "iBound=" << m_param_ibound << ","
391  << "Order=MinFill" << ","
392  << "Iter=" << m_param_iterations;
393  s.set_properties(oss.str());
394  s.run();
395  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
396  } else if (m_algorithm == MERLIN_ALGO_IJGP) {
397  merlin::ijgp s(fs);
398  std::ostringstream oss;
399  oss << "iBound=" << m_param_ibound << ","
400  << "Order=MinFill" << ","
401  << "Iter=" << m_param_iterations << ","
402  << "Task=MAP";
403  s.set_properties(oss.str());
404  s.run();
405  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
406  } else if (m_algorithm == MERLIN_ALGO_GIBBS) {
407  merlin::gibbs s(fs);
408  std::ostringstream oss;
409  oss << "TempMin=1.0,TempMax=1.0,Best=0,Beliefs=1" << ","
410  << "nIter=" << m_param_iterations << ","
411  << "nSamples=" << m_param_samples;
412  s.set_properties(oss.str());
413  s.run();
414  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
415  } else if (m_algorithm == MERLIN_ALGO_BE) {
416  merlin::bte s(fs);
417  std::ostringstream oss;
418  oss << "Order=MinFill" << ","
419  << "Task=MAP";
420  s.set_properties(oss.str());
421  std::vector<vindex> qvars;
422  for (size_t i = 0; i < gm.nvar(); ++i) {
423  if (m_evidence.find(i) == m_evidence.end()) {
424  size_t nvar = old2new.at(i);
425  qvars.push_back(nvar); // use the new index of the MAP vars
426  }
427  }
428  s.set_query(qvars);
429  s.run();
430  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
431  }
432  // follow with search-based AOBB, AOBF, RBFAOO
433  } else if ( m_task == MERLIN_TASK_MMAP ) {
434  size_t found = m_filename.find_last_of("/");
435  std::string prob_name = (found != std::string::npos) ?
436  m_filename.substr(found + 1) : m_filename;
437  std::string output_file = "./" + prob_name + ".MMAP";
438 
439  if (m_algorithm == MERLIN_ALGO_WMB) {
440  merlin::wmb s(fs);
441  std::ostringstream oss;
442  oss << "iBound=" << m_param_ibound << ","
443  << "Order=MinFill" << ","
444  << "OrderIter=100" << ","
445  << "Iter=" << m_param_iterations << ","
446  << "Task=MMAP";
447  s.set_properties(oss.str());
448  std::vector<size_t> qvars;
449  for (size_t i = 0; i < m_query.size(); ++i) {
450  vindex var = m_query[i];
451  vindex nvar = old2new.at(var);
452  qvars.push_back(nvar); // use the new index of the MAP vars
453  }
454  s.set_query(qvars);
455  s.run();
456  s.write_solution(output_file.c_str(), m_evidence, old2new, gm, m_first_solution);
457  } else if (m_algorithm == MERLIN_ALGO_BE) {
458  merlin::bte s(fs);
459  std::ostringstream oss;
460  oss << "Order=MinFill" << ","
461  << "Task=MMAP";
462  s.set_properties(oss.str());
463  std::vector<size_t> qvars;
464  for (size_t i = 0; i < m_query.size(); ++i) {
465  vindex var = m_query[i];
466  vindex nvar = old2new.at(var);
467  qvars.push_back(nvar); // use the new index of the MAP vars
468  }
469  s.set_query(qvars);
470  s.run();
471  s.write_solution(output_file.c_str(), m_evidence, old2new, gm);
472  }
473 
474  // TODO: follow with search-based AOBB, AOBF, RBFAOO
475  }
476 
477  if (m_first_solution) {
478  m_first_solution = false;
479  }
480 
481  return EXIT_SUCCESS;
482 
483  } catch (const std::runtime_error& e) {
484  std::cerr << e.what() << std::endl;
485  return EXIT_FAILURE;
486  }
