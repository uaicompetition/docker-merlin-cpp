478  double score(const vector<variable_set>& fin, const variable& VX, size_t i, size_t j) {
479  double err;
480  const variable_set& F1 = fin[i], &F2 = fin[j]; // (useful shorthand)
481  size_t iBound = std::max(std::max(m_ibound, F1.nvar() - 1),
482  F2.nvar() - 1); // always OK to keep same size
483  variable_set both = F1 + F2;
484  if (both.nvar() > iBound+1)
485  err = -3; // too large => -3
486  else
487  err = 1.0 / (F1.nvar() + F2.nvar()); // greedy scope-based 2 (check if useful???)
488  //else if (_byScope) err = 1; // scope-based => constant score
489  return err;
490  }
