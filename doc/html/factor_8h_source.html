1116  double distance(factor const& F2, Distance type = Distance::L2) const {
1117  assert(vars() == F2.vars());
1118  factor F(*this), Ftmp; // make a copy for manipulation
1119  double dist = -1.0; // local variables
1120  value Z;
1121  switch (type) {
1122  case Distance::L2: // L2, sum of squared errors
1123  F -= F2;
1124  F *= F;
1125  dist = F.sum();
1126  break;
1127  case Distance::L1: // L1, sum of absolute errors (=TV !!)
1128  F -= F2;
1129  dist = F.abs().sum();
1130  break;
1131  case Distance::LInf: // L-infinity, max absolute error
1132  F -= F2;
1133  dist = F.abs().max();
1134  break;
1135  case Distance::KL: // KL-divergence (relative entropy)
1136  Z = sum();
1137  F /= F2;
1138  F *= F2.sum() / Z;
1139  F.log();
1140  F *= *this;
1141  dist = F.sum() / Z;
1142  break;
1143  case Distance::HPM: // Hilbert's projective metric
1144  F /= F2;
1145  F.log();
1146  dist = F.max() - F.min(); // aka "dynamic range"
1147  break;
1148  case Distance::MAS: // "MAS" error value (not a metric)
1149  F.log();
1150  Ftmp = F2;
1151  F /= Ftmp.log();
1152  dist = std::max(F.max(), 1.0 / F.min()) - 1.0;
1153  break;
1154  case Distance::OptGap: // "Primal/Dual Gap"-like
1155  double mx1, mx2, gap1, gap2;
1156  mx1 = mx2 = gap1 = gap2 = 0.0;
1157  for (size_t i = 0; i < num_states(); ++i) {
1158  if (mx1 < F[i]) {
1159  gap2 = F2[i];
1160  mx1 = F[i];
1161  } else if (mx1 == F[i])
1162  gap2 = std::min(gap2, F2[i]);
1163  if (mx2 < F2[i]) {
1164  gap1 = F[i];
1165  mx2 = F2[i];
1166  } else if (mx2 == F2[i])
1167  gap1 = std::min(gap1, F[i]);
1168  }
1169  return (mx1 - gap1) + (mx2 - gap2);
1170  break;
1171  //case Distance::Hellinger: (!!)
1172  // F^=0.5; F-=F2^0.5; F*=F; dist=(0.5*F.sum())^0.5; // straightforward computation
1173  // F*=F2; F^=0.5; dist=(1-F.sum())^0.5; // alternate computation if F,F2 normalized
1174  // break;
1175  default:
1176  throw std::runtime_error("Invalid distance type");
1177  }
1178  return dist;
1179  };

1226  std::vector<factor> decomp_sum(std::vector<variable_set> vlist,
1227  factor::Decomp method) const {
1228  int nF = vlist.size();
1229  double mx, mn;
1230  std::vector<factor> Flist(nF);
1231 
1232  factor tmp, F = *this;
1233  switch (method) {
1234  case Decomp::L2: //L2
1235  double Cn, Cd;
1236  Cd = F.numel();
1237  Cn = F.sum(); // /Cd*(1-1.0/nF);
1238  for (int j = 0; j < nF; j++) {
1239  Flist[j] = F.marginal(vlist[j]);
1240  double D = Cd / Flist[j].numel();
1241  Flist[j] /= D;
1242  Flist[j] -= Cn / Cd * (1.0 - 1.0 / (nF - j));
1243  F -= Flist[j];
1244  Cn -= Flist[j].sum() * D;
1245  }
1246  break;
1247  case Decomp::L2_HPM: //L2+HPM
1248  Flist = decomp_sum(vlist, Decomp::L2);
1249  for (int j = 0; j < nF; j++)
1250  F -= Flist[j];
1251  mx = F.max();
1252  mn = F.min();
1253  for (int j = 0; j < nF; j++)
1254  Flist[j] += (mx + mn) / 2 / nF;
1255  break;
