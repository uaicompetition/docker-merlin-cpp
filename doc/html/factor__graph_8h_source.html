105  findex add_factor(const factor& F) {
106  findex use = graphical_model::add_factor(F); // add the factor to the underlying collection
107  m_vindex.resize(nvar(), vindex(-1)); // then update variable nodes and edges (!!!)
108  if (F.nvar() == 1 && m_vindex[*F.vars().begin()] == vindex(-1))
109  m_vindex[*F.vars().begin()] = use;
110  else
111  for (variable_set::const_iterator v = F.vars().begin();
112  v != F.vars().end(); ++v) {
113  if (m_vindex[*v] == vindex(-1))
114  m_vindex[*v] = graphical_model::add_factor(factor(*v, 1.0));
115  add_edge(use, local_factor(*v)); // add edge to variable node
116  }
117  return use;
118  }

226  void create_factor_graph() {
227 
228  // init var to node index map
229  if (m_vindex.size() < nvar()) {
230  m_vindex.resize(nvar());
231  }
232 
233  // identify unary factors (if any)
234  std::vector<bool> found(nvar(), false);
235  for (size_t i = 0; i < m_factors.size(); ++i) {
236  if (m_factors[i].nvar() == 1) {
237  size_t v = _vindex(*m_factors[i].vars().begin());
238  if (!found[v]) m_vindex[v] = i;
239  found[v] = true;
240  }
241  }
242 
243  // create local factors for each of the variables
244  for (size_t v = 0; v < found.size(); ++v) {
245  if (!found[v]) {
246  m_vindex[v] = graphical_model::add_factor(factor(var(v), 1.0));
247  }
248  }
249 
250  // create the bi-partite graph (edges from factor node to its variable nodes)
251  if (num_edges() != 0) {
252  throw std::runtime_error("Initial factor graph must be empty.");
253  }
254  for (size_t i = 0; i < m_factors.size(); ++i) {
255  if (!is_var_node(i)) { // factor node
256  for (variable_set::const_iterator v = m_factors[i].vars().begin();
257  v != m_factors[i].vars().end(); ++v) {
258  add_edge(i, local_factor(*v));
259  }
260  }
261  }
262  }

344  std::vector<edge_t> span_tree(TreeType tt, variable root) {
345  abstract_queue<findex> *Q;
346  switch (tt) {
347  case TreeType::WidthFirst:
348  Q = new abstract_queue_fifo<findex>();
349  break;
350  case TreeType::DepthFirst:
351  Q = new abstract_queue_lifo<findex>();
352  break;
353  case TreeType::MaxWeight:
354  default:
355  throw std::runtime_error("Not implemented");
356  break;
357  }
358  Q->push(local_factor(root));
359  std::vector<findex> used, fUsed;
360  used.resize(nvar(), findex(-1));
361  fUsed.resize(num_factors(), 0);
362  std::vector<edge_t> tree;
363  tree.reserve(2 * nvar());
364  while (!Q->empty()) {
365  findex next = Q->first();
366  Q->pop();
367  if (fUsed[next] != 0)
368  continue; // already taken this factor?
369  size_t nFound = 0, parent = 0;// if not look over variables and make sure
370  const variable_set& vs = get_factor(next).vars(); // at most one is already covered;
371  for (variable_set::const_iterator v = vs.begin(); v != vs.end(); ++v) {
372  if (used[_vindex(*v)] != vindex(-1)) {
373  parent = used[_vindex(*v)];
374  nFound++;
375  }
376  } // if one is covered that's our parent; more
377  if (nFound > 1)
378  continue;// means this factor is unavailable so keep looking
379  fUsed[next] = 1; // otherwise, mark factor, variables as covered
380  if (is_var_node(next))
381  used[_vindex(*vs.begin())] = next;// always cover by var node once visited
382  else
383  for (variable_set::const_iterator v = vs.begin(); v != vs.end();
384  ++v) {
385  if (used[_vindex(*v)] == vindex(-1))
386  used[_vindex(*v)] = next;// otherwise remember which factor we are
387  }
388  // if we have a parent, add an edge to our trees
389  if (nFound > 0)
390  tree.push_back(edge_t(parent, next)); //, _eindex(parent,next),_eindex(next,parent))); //!!!
391  // if this is a root, do nothing (!!!); don't keep track of isolated nodes?
392 
393  const set<edge_id> nbrs = neighbors(next);// add all neighbors to the search queue
394  for (set<edge_id>::const_iterator n = nbrs.begin(); n != nbrs.end();
395  ++n) {
396  Q->push(n->second);
397  }
398  }
399  delete Q;
400  return tree;
401  }
